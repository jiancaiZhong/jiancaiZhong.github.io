### 输入框动效

<div class="contain">
    <div class="input-box">
        <input type="text" required>
        <span>Your Name</span>
    </div>
</div>

<style>

.contain{
    display: flex;
    justify-content: center;
    align-items: center;
    height: 200px;
    width: 100%;
}

.input-box {
    position: relative;
    width: 250px;

}

.input-box input {
    width: 100%;
    border: 1px solid #ccc;
    border-radius: 8px;
    padding: 10px;
    font-size: 1em;
    outline: none;
}

.input-box span {
    position: absolute;
    left: 0;
    top: 0;
    padding: 10px;
    font-size: 1em;
    pointer-events: none;
    color: #ccc;
    transition: all 0.5s;
}

.input-box input:valid ~ span, .input-box input:focus ~ span {
    color: orange;
    transform: translateX(10px) translateY(-7px);
    font-size: 0.65em;
    padding: 0 10px;
    background-color: #fff;
}
